  5   glm::vec2 pixel_to_NDC(unsigned int x, unsigned int y) {
  6     glm::vec2 pixels((float)x, (float)y);
  7     pixels = pixels / glm::vec2(SCREEN_WIDTH, SCREEN_HEIGHT);
  8     pixels = (pixels * glm::vec2(2.0f)) - glm::vec2(1.0f);
  9     return pixels;
 10   }

 12   glm::vec2 pixel_to_font_size(unsigned int x, unsigned int y) {
 13     glm::vec2 pixels((float)x, (float)y);
 14     pixels = pixels / glm::vec2(SCREEN_WIDTH, SCREEN_HEIGHT);
 15     pixels = (pixels * glm::vec2(2.0f));
 16     return pixels;
 17   }

 19   glm::vec2 get_cursor_position(GLFWwindow *window) {
 20     int w, h;
 21     glfwGetWindowSize(window, &w, &h);
 22     //std::cout << "window size: " << w << " " << h << std::endl;
 23     double x, y;
 24     glfwGetCursorPos(window, &x, &y);
 25     //std::cout << "cursor position: " << x << " " << y << std::endl;
 26 
 27     return glm::vec2((2.0f * (x / w)) - 1.0f, 1.0f - (2.0f * (y / h)));
 28 
 29   }

 78   std::array<char_vertex, 4> Font::get_char_vertex_data(glm::vec2 screen_position, char character, glm::vec2 font_size) {
 79 
 80     //positions
 81     glm::vec2 pbl, pbr, ptl, ptr; // bottom left, bottom right, top left, top right
 82     pbl = screen_position;
 83     pbr = screen_position + glm::vec2(font_size.x, 0.0f);
 84     ptl = screen_position + glm::vec2(0.0f, font_size.y);
 85     ptr = screen_position + font_size;
 86 
 87     //texture coordinates
 88     glm::vec2 tbl, tbr, ttl, ttr;
 89     tbl = this->positions[character];
 90     tbr = tbl + glm::vec2(0.125f, 0.0f);
 91     ttl = tbl + glm::vec2(0.0f, 0.125f);
 92     ttr = tbl + glm::vec2(0.125f);
 93 
 94     std::array<char_vertex, 4> data;
 95     data[0] = (char_vertex){ .position = pbl, .tex_position = tbl };
 96     data[1] = (char_vertex){ .position = pbr, .tex_position = tbr };
 97     data[2] = (char_vertex){ .position = ptl, .tex_position = ttl };
 98     data[3] = (char_vertex){ .position = ptr, .tex_position = ttr };
 99 
100     return data;
101 
102   }

104   unsigned int Font::get_vao(glm::vec2 screen_position, char character, glm::vec2 font_size) {
105 
106     unsigned int VAO, VBO;
107     std::array<char_vertex, 4> vertices = get_char_vertex_data(screen_position, character, font_size);
108 
109     glGenVertexArrays(1, &VAO);
110     glGenBuffers(1, &VBO);
111 
112     glBindVertexArray(VAO);
113 
114     glBindBuffer(GL_ARRAY_BUFFER, VBO);
115     glBufferData(GL_ARRAY_BUFFER, sizeof(char_vertex) * vertices.size(), &vertices[0], GL_STATIC_DRAW);
116 
117     glEnableVertexAttribArray(0);
118     glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(char_vertex), (void*)0);
119     glEnableVertexAttribArray(1);
120     glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, sizeof(char_vertex), (void*)offsetof(char_vertex, tex_position));
121 
122     glBindVertexArray(0);
123 
124     return VAO;
125   }

127   Label::Label(glm::vec2 p, std::string t, Font f, glm::vec2 fs, glm::vec4 fc) : position(p), text(t), font(f), font_size(fs), color(fc) {
128     unsigned int line = 0;
129     unsigned int column = 0;
130     for (int i = 0; i<this->text.size(); i++) {
131       if (text[i] == '\n') {
132         line++;
133         column = 0;
134       } else {
135         this->character_VAOs.push_back(f.get_vao(this->position + glm::vec2(column * this->font_size.x, - (line * font_size.y)), text[i], font_size));
136         column++;
137       }
138     }
139   }

141   void Label::render() {
142     glActiveTexture(GL_TEXTURE1);
143     glBindTexture(GL_TEXTURE_2D, this->font.texture);
144     this->font.font_shader.use();
145     this->font.font_shader.set_uniform<int>("bitmap_font", 1);
146     this->font.font_shader.set_vector<glm::vec4>("font_color", this->color);
147     for (auto &vao : this->character_VAOs) {
148       glBindVertexArray(vao);
149       glDrawArrays(GL_TRIANGLE_STRIP, 0, 4);
150       glBindVertexArray(0);
151     }
152     glUseProgram(0);
153     glBindTexture(GL_TEXTURE_2D, 0);
154   }

156   Button::Button(glm::vec2 p, glm::vec2 s, std::function<void()> f) :
157     position(p), size(s), button_func(f),
158     button_shader(Shader("./shader/button_vertex.glsl", "./shader/button_fragment.glsl")) {
159 
160     std::array<glm::vec2, 8> button_vertex_data = {
161       p, glm::vec2(0.0f, 0.0f),
162       p + glm::vec2(s.x, 0.0f), glm::vec2(1.0f, 0.0f),
163       p + glm::vec2(0.0f, s.y), glm::vec2(0.0f, 1.0f),
164       p + s, glm::vec2(1.0f)
165     };
166 
167 
168     glGenVertexArrays(1, &this->VAO);
169     glGenBuffers(1, &this->VBO);
170 
171     glBindVertexArray(this->VAO);
172 
173     glBindBuffer(GL_ARRAY_BUFFER, this->VBO);
174     glBufferData(GL_ARRAY_BUFFER, sizeof(glm::vec2) * button_vertex_data.size(), &button_vertex_data[0], GL_STATIC_DRAW);
175 
176     glEnableVertexAttribArray(0);
177     glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(glm::vec2) * 2, (void*)0);
178     glEnableVertexAttribArray(1);
179     glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, sizeof(glm::vec2) * 2, (void*)(sizeof(glm::vec2)));
180 
181     glBindVertexArray(0);
182 
183   }

193   bool Button::contains(glm::vec2 current_cursor_position) {
194 
195     if (current_cursor_position.x >= this->position.x
196         && current_cursor_position.x <= this->position.x + this->size.x
197         && current_cursor_position.y >= this->position.y
198         && current_cursor_position.y <= this->position.y + this->size.y) {
199       return true;
200     } else {
201       return false;
202     }
203 
204   }
